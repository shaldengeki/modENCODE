def create
  user = User.find_by_email(params[:session][:email])
  if user and user.authenticate(params[:session][:password])
    sign_in user
    flash[:success] = "Welcome, " + user.name + "."
    redirect_to root_path
  else
    flash[:error] = 'Invalid email/password combination.'
    redirect_to root_path
  end
end
